\newcommand{\RefTri}{
	\draw[black, thick] (0,0) -- (0,3);
    \draw[black, thick] (0,3) -- (3,0);
    \draw[black, thick] (3,0) -- (0,0);
    \filldraw[black] (0,0) circle (3pt);
    \filldraw[black] (0,3) circle (3pt);
    \filldraw[black] (3,0) circle (3pt);
	}
\newcommand{\Vertices}[2]{
	\filldraw[black] \foreach \x in {#1} {(\x) circle (3pt)};
	\filldraw[blue] \foreach \x in {#2} {(\x) circle (3pt)};
	}
\newcommand{\IntTri}[2]{
	\draw[black,thick] (#1) \foreach \x in {#2} {-- (\x) } -- cycle;
	}
